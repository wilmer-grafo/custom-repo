--PROCESOS
SELECT *
FROM (SELECT ISNULL((SELECT cargo FROM usuario AS u WHERE u.id_user = ct.origen), 'ACAMARA')  AS cargoorigen,
             ISNULL((SELECT cargo FROM usuario AS u WHERE u.id_user = ct.destino), 'ACAMARA') AS cargodestino
      FROM control_tiempo AS ct
      WHERE id_exp LIKE 'ARIT-LPZ-%/2015') AS temp
GROUP BY temp.cargoorigen, temp.cargodestino
ORDER BY temp.cargoorigen, temp.cargodestino;
