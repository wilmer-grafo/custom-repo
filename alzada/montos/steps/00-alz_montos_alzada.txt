SELECT temp.C_Nro_Expediente, 1 AS id
FROM (SELECT LTRIM(RTRIM(ct.C_Nro_Expediente)) AS C_Nro_Expediente
      FROM TSS_MONTOS ct
      WHERE (LTRIM(RTRIM(ct.C_Nro_Expediente)) LIKE 'ARIT-CHQ-%/2021%' OR
             LTRIM(RTRIM(ct.C_Nro_Expediente)) LIKE 'ARIT-CHQ-%/2022%')
      GROUP BY LTRIM(RTRIM(ct.C_Nro_Expediente))) AS temp
ORDER BY temp.C_Nro_Expediente;
