EN SQL SERVER
=============
SELECT C_Nro_Expediente, LTRIM(RTRIM(C_Representante)) AS C_Representante
FROM TSS_RECURSOS_ALZADA
WHERE LEN(LTRIM(RTRIM(C_Representante))) > 2
AND C_Nro_Expediente LIKE 'ARIT-BEN%'
ORDER BY C_Nro_Expediente;

PEGAR TODOS LOS EXPEDIENTES
select numero_recurso_alzada, id from alzada.alz_recursos_alzada
where numero_recurso_alzada in (
    'ARIT-BEN-0001/2010',
'ARIT-BEN-0001/2012',
'ARIT-BEN-0001/2013',
'ARIT-BEN-0001/2014',
'ARIT-BEN-0001/2017'
